		function countToTen(){
			var exampleText="Example Ouput: ";
			var iterator = 1;
			var x = document.getElementById("exampleOutput")
			while (iterator <= 10) {
				exampleText+= iterator;
				if (iterator != 10) {
					exampleText+= ", ";
				}
				iterator++;
			}
			x.innerHTML = exampleText;
		}
