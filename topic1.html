Example 1: Topics - functions, while loop, if, local scope variables

This example counts from 1 to 10 and appends the number to a variable that is comma delimited. The example also uses a conditional "if" statement to decide when to use the comma delimiter.

Code Snippit


	<script>
		function countToTen(){
			// exampleText and iterator are scoped local to this function
			var exampleText="Example 1 Ouput: ";
			var iterator = 1;
			var x = document.getElementById("ex1Output")
			// loops until the iterator value is 11
			while (iterator <= 10) {
				exampleText+= iterator;
				// The if below allows ", " to be added the end of the exampleText variable until 
				// the iterator variable is 10 this makes sure there is not a trailing ", " in the final output
				if (iterator != 10) {
					exampleText+= ", ";
				}
				iterator++;
			}
			x.innerHTML = exampleText;
		}
	</script>

Example 2: Topics - functions, parameters, for loop, if/else

This example counts from 1 to 30 and outputs all even numbers on one line and odds on another. A for loop is used to do the counting and an if/else statment is used to decide if a number is even or odd and then writes the number to the correct variable.

Code Snippit


	<script>
		function countToThirty(){
			var evenNumbers = "";
			var oddNumbers = "";
			// for loop will loop 30 times and add the iterators string value to the correct variable
			for(i=1; i < 31; i++){
				if( i % 2 == 0) {
					evenNumbers += i + ", ";
				} else {
					oddNumbers += i + ", ";
				}
			}
			// Demonstrates passing arguments to a function and getting a return value. 
			var ex2OutputHtml = writeEvenOddValues(evenNumbers,oddNumbers)
			document.getElementById("ex2Output").innerHTML = ex2OutputHtml;
		}

		// this demonstrates a functions with parameters
		// the functions takes the passed values and constructs html to be assigned to the innerHTML of a element
		function writeEvenOddValues(evenNumbers, oddNumbers){
			evenNumbers = evenNumbers.slice(0,-2);
			oddNumbers = oddNumbers.slice(0,-2);
			var ex2OutputHtml = "Even Numbers 1 to 30: " + evenNumbers + "<br>";
			ex2OutputHtml += "Odd Numbers 1 to 30: " + oddNumbers;
			return ex2OutputHtml;
		}
	</script>

Example 3: Topics - function, Array, Object, for, switch, for in

This example displays Lavold family members and their favorite foods. Empty array and object variables are declared and then food items in family member order are pushed into the arry. The array is then iterated through and using a switch statement food and names are matched together as key/value pair and added to the empty favFoods object. The object is then iterated through using a For/in loop to output the persons name and their favorite food.

Code Snippit


	<script>
		function favFoods(){
			// declare empty foods array and empty favFoods object
			var foods = [];
			var favFoods = {};
			var htmlOutput = "Lavold Family Favorite Foods List";

			// push/add 5 food items to the end of the array
			foods.push("Spaghetti");
			foods.push("Blue Bell Icecream");
			foods.push("Pizza");
			foods.push("Diet Coke");
			foods.push("Thai Curry");

			// for iterates through array elements and the switch adds key/value pairs to the favFoods object
			for (i=0 ; i < foods.length ; i++ ){
				switch (i){
					case 0:
						favFoods.Aaron = foods[i];
						break;
					case 1:
						favFoods.Liz = foods[i];
						break;
					case 2:
						favFoods.Megan = foods[i];
						break;
					case 3:
						favFoods.Jake = foods[i];
						break;
					case 4:
						favFoods.Kaelen = foods[i];
						break;
				}
			}

			// for/in statement iterates through the favFoods object and outputs html of favorite foods
			for (person in favFoods){
				htmlOutput += "<br>" + person + ": " + favFoods[person] + ".";
			}
			document.getElementById("ex3Output").innerHTML = htmlOutput;
		}
	</script>
